import { useRef, useEffect } from 'react';
import { useThree } from '@react-three/fiber';
import type { StarPoint } from '../types';
 
interface StarHoverPreviewProps {
  hoveredStar: StarPoint | null;
  mouse: { x: number; y: number };
}
 
export const StarHoverPreview = ({ hoveredStar, mouse }: StarHoverPreviewProps) => {
  const { gl } = useThree();
  const previewRef = useRef<HTMLDivElement>(null);
  
  useEffect(() => {
    if (hoveredStar && previewRef.current && gl) {
      // 获取鼠标在屏幕上的位置
      const mouseX = mouse.x * gl.domElement.clientWidth / 2 + gl.domElement.clientWidth / 2;
      const mouseY = -mouse.y * gl.domElement.clientHeight / 2 + gl.domElement.clientHeight / 2;
      
      // 设置预览框位置，避免超出屏幕
      const previewWidth = 200;
      const previewHeight = 100;
      const x = Math.min(mouseX + 10, gl.domElement.clientWidth - previewWidth);
      const y = Math.min(mouseY + 10, gl.domElement.clientHeight - previewHeight);
      
      previewRef.current.style.left = `${x}px`;
      previewRef.current.style.top = `${y}px`;
    }
  }, [hoveredStar, mouse, gl]);
  
  if (!hoveredStar) return null;
  
  return (
    <div ref={previewRef} className="star-hover-preview">
      <h4>{hoveredStar.title || '未命名'}</h4>
      <p className="preview-author">作者: {(hoveredStar.author || []).join(', ') || '未知'}</p>
      <p className="preview-tags">
        {(hoveredStar.tags || []).slice(0, 3).map((tag, index) => (
          <span key={index} className="preview-tag">{tag}</span>
        ))}
      </p>
    </div>
  );
};
