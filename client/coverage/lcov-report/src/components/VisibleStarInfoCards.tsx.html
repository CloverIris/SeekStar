import { useThree } from '@react-three/fiber';
import * as THREE from 'three';
import type { StarPoint } from '../types';
 
interface VisibleStarInfoCardsProps {
  visibleStars: Array<{ star: StarPoint; lod: number }>;
}
 
export const VisibleStarInfoCards = ({ visibleStars }: VisibleStarInfoCardsProps) => {
  const { gl, camera } = useThree();
  
  if (!visibleStars || visibleStars.length === 0) return null;
  
  return (
    <>
      {visibleStars.map(({ star, lod }) => {
        if (!star || !star.position) return null;
        
        // 计算星点在屏幕上的位置
        const starPosition = new THREE.Vector3(
          star.position.x,
          star.position.y,
          star.position.z
        );
        const screenPos = starPosition.clone().project(camera);
        
        if (!gl) return null;
        
        // 转换为CSS像素坐标
        const x = screenPos.x * gl.domElement.clientWidth / 2 + gl.domElement.clientWidth / 2;
        const y = -screenPos.y * gl.domElement.clientHeight / 2 + gl.domElement.clientHeight / 2;
        
        // 根据LOD级别选择要显示的信息模板
        let cardContent;
        if (lod === 3) {
          // LOD级别3：显示完整信息
          cardContent = (
            <div className="star-info-card lod-3">
              <h4>{star.title || '未命名'}</h4>
              <p className="star-author">作者: {(star.author || []).join(', ') || '未知'}</p>
              <p className="star-content">{star.content || '无内容'}</p>
              <div className="star-tags">
                {(star.tags || []).slice(0, 3).map((tag, index) => (
                  <span key={index} className="star-tag">{tag || ''}</span>
                ))}
              </div>
            </div>
          );
        } else if (lod === 2) {
          // LOD级别2：显示标题和作者
          cardContent = (
            <div className="star-info-card lod-2">
              <h4>{star.title || '未命名'}</h4>
              <p className="star-author">作者: {(star.author || []).join(', ') || '未知'}</p>
            </div>
          );
        } else {
          // LOD级别1：只显示标题
          cardContent = (
            <div className="star-info-card lod-1">
              <h4>{star.title || '未命名'}</h4>
            </div>
          );
        }
        
        return (
          <div 
            key={star.id} 
            className="star-info-card-container"
            style={{
              left: `${x}px`,
              top: `${y}px`,
              transform: 'translate(-50%, -100%)', // 居中对齐星点
              position: 'absolute',
              pointerEvents: 'none', // 确保不影响星点交互
              zIndex: 500
            }}
          >
            {cardContent}
          </div>
        );
      })}
    </>
  );
};
